struct MyConfig {
  // --- Architecture ---
  int inputSize;
  int numClasses;     // always 2 for regression (dist + conf output neurons)
  int conv1Filters;
  int conv2Filters;

  // --- Derived ---
  int conv1OutputSize;
  int pool1OutputSize;
  int conv2OutputSize;
  int flattenedSize;
  int conv1Weights;
  int conv2Weights;
  int outputWeights;

  // --- Training ---
  float learningRate;
  int   batchSize;
  int   targetEpochs;
  int   thresholdPress;
  int   thresholdRelease;
  unsigned long screenTimeout;
  int   numLabels;
  String classLabels[8];  // e.g. "0Blank","1Near_2cm","2Mid_5cm","3Far_10cm"
  String weightsFile;
  int   version;
  int   minVersion;
  bool  useAugmentation;
  bool  useGrayscale;
  bool  imagesToPsram;
  int   validationImages;
  float dropoutRate;

  // [INJECT-3c] Regression-specific config fields
  float distMax;          // max real-world distance (cm) — scales myDistPred
  int   numDistClasses;   // number of non-blank labelled distance classes
  float confThreshold;    // confidence threshold for "object detected"
  float confLossWeight;   // weight of BCE confidence loss vs MSE distance loss
};

MyConfig myCfg;

void myComputeArchSizes() {
  // Standard 2D conv sizes — correct for camera regression model.
  myCfg.conv1OutputSize = myCfg.inputSize - 2;
  myCfg.pool1OutputSize = myCfg.conv1OutputSize / 2;
  myCfg.conv2OutputSize = myCfg.pool1OutputSize - 2;
  myCfg.flattenedSize   = myCfg.conv2OutputSize * myCfg.conv2OutputSize * myCfg.conv2Filters;
  int ch = myCfg.useGrayscale ? 1 : 3;
  myCfg.conv1Weights  = 3 * 3 * ch * myCfg.conv1Filters;
  myCfg.conv2Weights  = 3 * 3 * myCfg.conv1Filters * myCfg.conv2Filters;
  myCfg.outputWeights = myCfg.flattenedSize * myCfg.numClasses; // flattenedSize × 2
}

void myApplyDefaultConfig() {
  myCfg.inputSize        = DEFAULT_INPUT_SIZE;
  myCfg.numClasses       = DEFAULT_NUM_CLASSES;
  myCfg.conv1Filters     = DEFAULT_CONV1_FILTERS;
  myCfg.conv2Filters     = DEFAULT_CONV2_FILTERS;
  myCfg.learningRate     = DEFAULT_LEARNING_RATE;
  myCfg.batchSize        = DEFAULT_BATCH_SIZE;
  myCfg.targetEpochs     = DEFAULT_TARGET_EPOCHS;
  myCfg.thresholdPress   = DEFAULT_THRESHOLD_PRESS;
  myCfg.thresholdRelease = DEFAULT_THRESHOLD_RELEASE;
  myCfg.screenTimeout    = DEFAULT_SCREEN_TIMEOUT;
  myCfg.weightsFile      = DEFAULT_WEIGHTS_FILE;
  myCfg.numLabels        = DEFAULT_NUM_LABELS;
  for (int i = 0; i < DEFAULT_NUM_LABELS; i++)
    myCfg.classLabels[i] = String(DEFAULT_LABELS[i]);
  myCfg.version          = CURRENT_VERSION;
  myCfg.minVersion       = MIN_VERSION;
  myCfg.useAugmentation  = DEFAULT_USE_AUGMENTATION;
  myCfg.useGrayscale     = DEFAULT_USE_GRAYSCALE;
  myCfg.imagesToPsram    = DEFAULT_IMAGES_TO_PSRAM;
  myCfg.validationImages = DEFAULT_VALIDATION_IMAGES;
  myCfg.dropoutRate      = DEFAULT_DROPOUT_RATE;
  // [INJECT-3d] Regression-specific defaults
  myCfg.distMax         = DEFAULT_DIST_MAX;
  myCfg.numDistClasses  = DEFAULT_NUM_DIST_CLASSES;
  myCfg.confThreshold   = DEFAULT_CONF_THRESHOLD;
  myCfg.confLossWeight  = DEFAULT_CONF_LOSS_WEIGHT;
  myComputeArchSizes();
}

void mySaveConfig() {
  if (!SD.exists("/header")) SD.mkdir("/header");
  File f = SD.open("/header/config.json", FILE_WRITE);
  if (!f) { Serial.println("[config] ERROR: could not write config.json"); return; }
  JsonDocument doc;
  doc["inputSize"]         = myCfg.inputSize;
  doc["numClasses"]        = myCfg.numClasses;
  doc["conv1Filters"]      = myCfg.conv1Filters;
  doc["conv2Filters"]      = myCfg.conv2Filters;
  doc["learningRate"]      = myCfg.learningRate;
  doc["batchSize"]         = myCfg.batchSize;
  doc["targetEpochs"]      = myCfg.targetEpochs;
  doc["thresholdPress"]    = myCfg.thresholdPress;
  doc["thresholdRelease"]  = myCfg.thresholdRelease;
  doc["screenTimeout"]     = (unsigned long)myCfg.screenTimeout;
  doc["weightsFile"]       = myCfg.weightsFile;
  doc["version"]           = myCfg.version;
  doc["minVersion"]        = myCfg.minVersion;
  doc["useAugmentation"]   = myCfg.useAugmentation;
  doc["useGrayscale"]      = myCfg.useGrayscale;
  doc["imagesToPsram"]     = myCfg.imagesToPsram;
  doc["validationImages"]  = myCfg.validationImages;
  doc["dropoutRate"]       = myCfg.dropoutRate;
  // [INJECT-3e] Regression-specific serialisation
  doc["distMax"]           = myCfg.distMax;
  doc["numDistClasses"]    = myCfg.numDistClasses;
  doc["confThreshold"]     = myCfg.confThreshold;
  doc["confLossWeight"]    = myCfg.confLossWeight;
  JsonArray labels = doc["classLabels"].to<JsonArray>();
  for (int i = 0; i < myCfg.numLabels; i++) labels.add(myCfg.classLabels[i]);
  serializeJsonPretty(doc, f);
  f.close();
  Serial.println("[config] Wrote config to /header/config.json");
}

void myLoadConfig() {
  myApplyDefaultConfig();
  const char* path = "/header/config.json";
  if (!SD.exists(path)) { Serial.println("[config] config.json not found — using defaults"); mySaveConfig(); return; }
  File f = SD.open(path, FILE_READ);
  if (!f) { Serial.println("[config] ERROR: could not open config.json"); return; }
  JsonDocument doc;
  DeserializationError err = deserializeJson(doc, f);
  f.close();
  if (err) { Serial.printf("[config] JSON parse error: %s — using defaults\n", err.c_str()); return; }

  myCfg.inputSize        = doc["inputSize"]        | DEFAULT_INPUT_SIZE;
  myCfg.numClasses       = doc["numClasses"]       | DEFAULT_NUM_CLASSES;
  myCfg.conv1Filters     = doc["conv1Filters"]     | DEFAULT_CONV1_FILTERS;
  myCfg.conv2Filters     = doc["conv2Filters"]     | DEFAULT_CONV2_FILTERS;
  myCfg.learningRate     = doc["learningRate"]      | DEFAULT_LEARNING_RATE;
  myCfg.batchSize        = doc["batchSize"]        | DEFAULT_BATCH_SIZE;
  myCfg.targetEpochs     = doc["targetEpochs"]     | DEFAULT_TARGET_EPOCHS;
  myCfg.thresholdPress   = doc["thresholdPress"]   | DEFAULT_THRESHOLD_PRESS;
  myCfg.thresholdRelease = doc["thresholdRelease"] | DEFAULT_THRESHOLD_RELEASE;
  myCfg.screenTimeout    = doc["screenTimeout"]    | (int)DEFAULT_SCREEN_TIMEOUT;
  myCfg.weightsFile      = doc["weightsFile"]      | DEFAULT_WEIGHTS_FILE;
  myCfg.version          = doc["version"]          | CURRENT_VERSION;
  myCfg.minVersion       = doc["minVersion"]       | MIN_VERSION;
  myCfg.useAugmentation  = doc["useAugmentation"]  | DEFAULT_USE_AUGMENTATION;
  myCfg.useGrayscale     = doc["useGrayscale"]     | DEFAULT_USE_GRAYSCALE;
  myCfg.imagesToPsram    = doc["imagesToPsram"]    | DEFAULT_IMAGES_TO_PSRAM;
  myCfg.validationImages = doc["validationImages"] | DEFAULT_VALIDATION_IMAGES;
  myCfg.dropoutRate      = doc["dropoutRate"]      | DEFAULT_DROPOUT_RATE;
  // [INJECT-3f] Regression-specific deserialisation
  myCfg.distMax          = doc["distMax"]          | DEFAULT_DIST_MAX;
  myCfg.numDistClasses   = doc["numDistClasses"]   | DEFAULT_NUM_DIST_CLASSES;
  myCfg.confThreshold    = doc["confThreshold"]    | DEFAULT_CONF_THRESHOLD;
  myCfg.confLossWeight   = doc["confLossWeight"]   | DEFAULT_CONF_LOSS_WEIGHT;

  if (myCfg.version < MIN_VERSION) {
    Serial.printf("[config] ERROR: version %d < min %d\n", myCfg.version, MIN_VERSION);
    u8g2.firstPage();
    do { u8g2.drawStr(0,10,"CONFIG ERR"); u8g2.drawStr(0,22,"Old version"); } while(u8g2.nextPage());
    while(true) { delay(1000); }
  }
  if (doc["classLabels"].is<JsonArray>()) {
    JsonArray arr = doc["classLabels"].as<JsonArray>();
    int count = 0;
    for (int ai = 0; ai < (int)arr.size() && count < 8; ai++) { myCfg.classLabels[count++] = arr[ai].as<String>(); }
    if (count > 0) { myCfg.numLabels = count; }
    // numClasses stays 2 (output neurons) regardless of label count
  }
  myComputeArchSizes();
  Serial.printf("[config] Loaded: inputSize=%d numClasses=%d flat=%d distMax=%.1f\n",
    myCfg.inputSize, myCfg.numClasses, myCfg.flattenedSize, myCfg.distMax);
}

void myResetTouchState() { myTouchState=0; myTouchTimer=0; myTouchTapCount=0; myTouchPending=false; }

int myReadTouch() { return analogRead(A0); }

void myUpdateTouchState() {
  int raw = myReadTouch();
  if (raw > myThresholdPress)  { if (myTouchState==0) { myTouchState=1; myTouchTimer=millis(); } }
  if (raw < myThresholdRelease){ if (myTouchState==1) { myTouchState=0; myTouchTapCount++; myTouchPending=true; myTouchTimer=millis(); } }
}

int myCheckTouchInput() {
  myUpdateTouchState();
  if (myTouchPending && (millis()-myTouchTimer > 250)) {
    int taps = myTouchTapCount; myTouchTapCount=0; myTouchPending=false;
    if (taps == 1) return 1;
    if (taps >= 3) return 2;
  }
  return 0;
}

int myPeekTouchAction() {
  myUpdateTouchState();
  if (myTouchPending && (millis()-myTouchTimer > 250)) {
    if (myTouchTapCount == 1) return 1;
    if (myTouchTapCount >= 3) return 2;
  }
  return 0;
}

// Inline helpers replace macros — avoids #define-inside-function
// parser confusion on xtensa-g++ (ESP32 Arduino gcc 8.4).
static void myReallocBuf(float** ptr, size_t bytes) {
  if (*ptr) { free(*ptr); *ptr = nullptr; }
  *ptr = (float*)ps_malloc(bytes);
}
static void myReallocBufZero(float** ptr, size_t bytes) {
  if (*ptr) { free(*ptr); *ptr = nullptr; }
  *ptr = (float*)ps_malloc(bytes);
  if (*ptr) memset(*ptr, 0, bytes);
}

void myAllocateMemory(bool resetMoments = true) {
  myReallocBuf(&myConv1_w,      CONV1_WEIGHTS  * sizeof(float));
  myReallocBuf(&myConv1_b,      CONV1_FILTERS  * sizeof(float));
  myReallocBuf(&myConv2_w,      CONV2_WEIGHTS  * sizeof(float));
  myReallocBuf(&myConv2_b,      CONV2_FILTERS  * sizeof(float));
  myReallocBuf(&myOutput_w,     OUTPUT_WEIGHTS * sizeof(float));
  myReallocBuf(&myOutput_b,     NUM_CLASSES    * sizeof(float));
  myReallocBuf(&myConv1_w_grad, CONV1_WEIGHTS  * sizeof(float));
  myReallocBuf(&myConv1_b_grad, CONV1_FILTERS  * sizeof(float));
  myReallocBuf(&myConv2_w_grad, CONV2_WEIGHTS  * sizeof(float));
  myReallocBuf(&myConv2_b_grad, CONV2_FILTERS  * sizeof(float));
  myReallocBuf(&myOutput_w_grad,OUTPUT_WEIGHTS * sizeof(float));
  myReallocBuf(&myOutput_b_grad,NUM_CLASSES    * sizeof(float));
  myReallocBuf(&myConv1_output, CONV1_OUTPUT_SIZE*CONV1_OUTPUT_SIZE*CONV1_FILTERS*sizeof(float));
  myReallocBuf(&myPool1_output, POOL1_OUTPUT_SIZE*POOL1_OUTPUT_SIZE*CONV1_FILTERS*sizeof(float));
  myReallocBuf(&myConv2_output, CONV2_OUTPUT_SIZE*CONV2_OUTPUT_SIZE*CONV2_FILTERS*sizeof(float));
  myReallocBuf(&myDense_output, NUM_CLASSES    * sizeof(float));
  myReallocBuf(&myDense_grad,   FLATTENED_SIZE * sizeof(float));
  myReallocBuf(&myConv2_grad,   CONV2_OUTPUT_SIZE*CONV2_OUTPUT_SIZE*CONV2_FILTERS*sizeof(float));
  myReallocBuf(&myPool1_grad,   POOL1_OUTPUT_SIZE*POOL1_OUTPUT_SIZE*CONV1_FILTERS*sizeof(float));
  myReallocBuf(&myConv1_grad,   CONV1_OUTPUT_SIZE*CONV1_OUTPUT_SIZE*CONV1_FILTERS*sizeof(float));
  myReallocBuf(&myInputBuffer,  INPUT_SIZE*INPUT_SIZE*INPUT_CHANNELS*sizeof(float));
  myReallocBuf(&myDropoutMask,  FLATTENED_SIZE * sizeof(float));

  if (resetMoments) {
    myReallocBufZero(&myConv1_w_m, CONV1_WEIGHTS*sizeof(float));
    myReallocBufZero(&myConv1_w_v, CONV1_WEIGHTS*sizeof(float));
    myReallocBufZero(&myConv1_b_m, CONV1_FILTERS*sizeof(float));
    myReallocBufZero(&myConv1_b_v, CONV1_FILTERS*sizeof(float));
    myReallocBufZero(&myConv2_w_m, CONV2_WEIGHTS*sizeof(float));
    myReallocBufZero(&myConv2_w_v, CONV2_WEIGHTS*sizeof(float));
    myReallocBufZero(&myConv2_b_m, CONV2_FILTERS*sizeof(float));
    myReallocBufZero(&myConv2_b_v, CONV2_FILTERS*sizeof(float));
    myReallocBufZero(&myOutput_w_m, OUTPUT_WEIGHTS*sizeof(float));
    myReallocBufZero(&myOutput_w_v, OUTPUT_WEIGHTS*sizeof(float));
    myReallocBufZero(&myOutput_b_m, NUM_CLASSES*sizeof(float));
    myReallocBufZero(&myOutput_b_v, NUM_CLASSES*sizeof(float));
  }

  if (myConv1_w_grad) memset(myConv1_w_grad,  0, CONV1_WEIGHTS*sizeof(float));
  if (myConv1_b_grad) memset(myConv1_b_grad,  0, CONV1_FILTERS*sizeof(float));
  if (myConv2_w_grad) memset(myConv2_w_grad,  0, CONV2_WEIGHTS*sizeof(float));
  if (myConv2_b_grad) memset(myConv2_b_grad,  0, CONV2_FILTERS*sizeof(float));
  if (myOutput_w_grad) memset(myOutput_w_grad, 0, OUTPUT_WEIGHTS*sizeof(float));
  if (myOutput_b_grad) memset(myOutput_b_grad, 0, NUM_CLASSES*sizeof(float));

  if (myConv1_w)  for(int i=0;i<CONV1_WEIGHTS; i++) myConv1_w[i]  = ((float)random(-1000,1000))/100000.f;
  if (myConv2_w)  for(int i=0;i<CONV2_WEIGHTS; i++) myConv2_w[i]  = ((float)random(-1000,1000))/100000.f;
  if (myOutput_w) for(int i=0;i<OUTPUT_WEIGHTS;i++) myOutput_w[i] = ((float)random(-1000,1000))/100000.f;
  if (myConv1_b)  memset(myConv1_b,  0, CONV1_FILTERS*sizeof(float));
  if (myConv2_b)  memset(myConv2_b,  0, CONV2_FILTERS*sizeof(float));
  if (myOutput_b) {
    memset(myOutput_b, 0, NUM_CLASSES*sizeof(float));
    myOutput_b[0] = 0.9f;  // v003 CHANGE-2: warm-start distance bias near top target
    myOutput_b[1] = 0.0f;  // confidence bias starts at zero
  }

  Serial.printf("[mem] Alloc done. Free PSRAM: %d bytes\n", ESP.getFreePsram());
}

void mySaveWeights() {
  if (!mySDavailable) { Serial.println("[weights] No SD — cannot save"); return; }
  String filePath = "/header/" + myCfg.weightsFile;
  if (!SD.exists("/header")) SD.mkdir("/header");
  File f = SD.open(filePath.c_str(), FILE_WRITE);
  if (!f) { Serial.println("[weights] ERROR: cannot open for write"); return; }

  f.println("--- WEIGHTS HEADER BEGIN ---");
  JsonDocument doc;
  doc["version"]      = CURRENT_VERSION;
  doc["inputSize"]    = myCfg.inputSize;

bool myLoadWeights(bool resetMoments = true) {
  if (!mySDavailable) { Serial.println("[weights] No SD — skipping load"); return false; }
  String filePath = "/header/" + myCfg.weightsFile;
  if (!SD.exists(filePath.c_str())) { Serial.println("[weights] No weights file on SD"); return false; }
  File f = SD.open(filePath.c_str(), FILE_READ);
  if (!f) return false;

  String line;
  bool inHeader = false, pastHeader = false;
  while (f.available() && !pastHeader) {
    line = f.readStringUntil('\n'); line.trim();
    if (line == "--- WEIGHTS HEADER BEGIN ---") inHeader = true;
    if (inHeader && line == "--- WEIGHTS HEADER END ---") {
      f.readStringUntil('\n');
      pastHeader = true;
    }
  }
  if (!pastHeader) { f.close(); Serial.println("[weights] Bad header"); return false; }

  f.read((uint8_t*)myConv1_w,  CONV1_WEIGHTS *4);
  f.read((uint8_t*)myConv1_b,  CONV1_FILTERS *4);
  f.read((uint8_t*)myConv2_w,  CONV2_WEIGHTS *4);
  f.read((uint8_t*)myConv2_b,  CONV2_FILTERS *4);
  f.read((uint8_t*)myOutput_w, OUTPUT_WEIGHTS*4);
  f.read((uint8_t*)myOutput_b, NUM_CLASSES   *4);
  // v003 OPTIONAL-3e: restore calibration
  if (f.read((uint8_t*)&myCalibSlope,  sizeof(float)) == sizeof(float) &&
      f.read((uint8_t*)&myCalibOffset, sizeof(float)) == sizeof(float)) {
    myCalibReady = true;
    Serial.printf("[weights] Calibration loaded: slope=%.3f offset=%.3f\n", myCalibSlope, myCalibOffset);
  } else {
    myCalibReady = false;  // fallback to raw * distMax
  }
  f.close();
  myWeightsTrained = true;
  Serial.println("[weights] Loaded from SD successfully");
  return true;
}

bool myLoadImageFromFile(const char* path, float* buf) {
  if (!myRgbBuffer) return false;
  File f = SD.open(path);
  if (!f) return false;
  size_t sz = f.size();
  uint8_t* jpg = (uint8_t*)ps_malloc(sz);
  if (!jpg) { f.close(); return false; }
  f.read(jpg, sz); f.close();
  bool ok = fmt2rgb888(jpg, sz, PIXFORMAT_JPEG, myRgbBuffer);
  free(jpg);
  if (!ok) return false;
  for (int y=0;y<INPUT_SIZE;y++) {
    for (int x=0;x<INPUT_SIZE;x++) {
      int sy = (int)((y+0.5)*240.0/INPUT_SIZE); if(sy>239) sy=239;
      int sx = (int)((x+0.5)*240.0/INPUT_SIZE); if(sx>239) sx=239;
      int srcIdx = (sy*240+sx)*3;
      int dstIdx = (y*INPUT_SIZE+x)*3;
      buf[dstIdx]   = myRgbBuffer[srcIdx]   / 255.0f;
      buf[dstIdx+1] = myRgbBuffer[srcIdx+1] / 255.0f;
      buf[dstIdx+2] = myRgbBuffer[srcIdx+2] / 255.0f;
    }
  }
  return true;
}

// ══════════════════════════════════════════════════════════════════
// PART 7: ML CORE — [INJECT-6]
// Full regression forward/backward from on-device-regression v003.
// WebSerial hooks added per PROMPT02.md spec.
// ══════════════════════════════════════════════════════════════════

// ─── Forward pass ────────────────────────────────────────────────
// Input: myInputBuffer (INPUT_SIZE×INPUT_SIZE×INPUT_CHANNELS)
// Output: myDense_output[0]=myDistPred (linear, can exceed 1.0)
//         myDense_output[1]=myConfPred (sigmoid [0..1])

void myForwardPass(float* input) {
  // Conv1: 3×3 valid, ReLU, depth-first layout
  for (int f=0; f<CONV1_FILTERS; f++) {
    float b = myConv1_b[f];
    for (int y=0; y<CONV1_OUTPUT_SIZE; y++) {
      for (int x=0; x<CONV1_OUTPUT_SIZE; x++) {
        float acc = b;
        for (int ky=0; ky<3; ky++) {
          for (int kx=0; kx<3; kx++) {
            int sy=y+ky, sx=x+kx;
            for (int c=0; c<INPUT_CHANNELS; c++) {
              int wIdx = ((ky*3+kx)*INPUT_CHANNELS+c)*CONV1_FILTERS + f;
              acc += input[(sy*INPUT_SIZE+sx)*INPUT_CHANNELS+c] * myConv1_w[wIdx];
            }
          }
        }
        int outIdx = (y*CONV1_OUTPUT_SIZE+x)*CONV1_FILTERS + f;
        myConv1_output[outIdx] = acc > 0 ? acc : 0;
      }
    }
  }

  // Pool1: 2×2 max pooling
  for (int f=0; f<CONV1_FILTERS; f++) {
    for (int y=0; y<POOL1_OUTPUT_SIZE; y++) {
      for (int x=0; x<POOL1_OUTPUT_SIZE; x++) {
        float mx = -1e9f;
        for (int py=0; py<2; py++)
          for (int px=0; px<2; px++) {
            int sy=y*2+py, sx=x*2+px;
            float v = myConv1_output[(sy*CONV1_OUTPUT_SIZE+sx)*CONV1_FILTERS+f];
            if (v > mx) mx = v;
          }
        myPool1_output[(y*POOL1_OUTPUT_SIZE+x)*CONV1_FILTERS+f] = mx;
      }
    }
  }

  // Conv2: 3×3 valid, ReLU
  for (int f=0; f<CONV2_FILTERS; f++) {
    float b = myConv2_b[f];
    for (int y=0; y<CONV2_OUTPUT_SIZE; y++) {
      for (int x=0; x<CONV2_OUTPUT_SIZE; x++) {
        float acc = b;
        for (int ky=0; ky<3; ky++) {
          for (int kx=0; kx<3; kx++) {
            int sy=y+ky, sx=x+kx;
            for (int inf=0; inf<CONV1_FILTERS; inf++) {
              int wIdx = ((ky*3+kx)*CONV1_FILTERS+inf)*CONV2_FILTERS + f;
              acc += myPool1_output[(sy*POOL1_OUTPUT_SIZE+sx)*CONV1_FILTERS+inf] * myConv2_w[wIdx];
            }
          }
        }
        int outIdx = (y*CONV2_OUTPUT_SIZE+x)*CONV2_FILTERS + f;
        myConv2_output[outIdx] = acc > 0 ? acc : 0;
      }
    }
  }

  // Flatten → output (2 neurons: dist=linear, conf=sigmoid)
  // Distance neuron (index 0): linear — no activation
  // Confidence neuron (index 1): sigmoid
  for (int n=0; n<NUM_CLASSES; n++) {
    float acc = myOutput_b[n];
    int flat = CONV2_OUTPUT_SIZE*CONV2_OUTPUT_SIZE*CONV2_FILTERS;
    for (int i=0; i<flat; i++) {
      // Weight layout: w[input_idx * NUM_CLASSES + output_idx]
      acc += myConv2_output[i] * myOutput_w[i * NUM_CLASSES + n];
    }
    if (n == 1) acc = 1.0f / (1.0f + expf(-acc));  // sigmoid for confidence
    myDense_output[n] = acc;
  }

  myDistPred = myDense_output[0];
  myConfPred = myDense_output[1];
}

// ─── Calibration fit (v003 CHANGE-3b) ─────────────────────────────
void myFitCalibration(float myRawPreds[], float myRealDists[], int myN) {
  if (myN < 2) {
    myCalibSlope = myDistMax; myCalibOffset = 0.0f; myCalibReady = true;
    Serial.println("Calibration: fallback identity");
    return;
  }
  float mySumX=0, mySumY=0, mySumXX=0, mySumXY=0;
  for (int i=0; i<myN; i++) {
    mySumX  += myRawPreds[i];
    mySumY  += myRealDists[i];
    mySumXX += myRawPreds[i] * myRawPreds[i];
    mySumXY += myRawPreds[i] * myRealDists[i];
  }
  float myDenom = myN * mySumXX - mySumX * mySumX;
  if (fabsf(myDenom) < 1e-6f) {
    myCalibSlope  = (mySumX > 1e-4f) ? mySumY / mySumX : myDistMax;
    myCalibOffset = 0.0f;
  } else {
    myCalibSlope  = (myN * mySumXY - mySumX * mySumY) / myDenom;
    myCalibOffset = (mySumY - myCalibSlope * mySumX) / myN;
  }
  myCalibReady = true;
  Serial.printf("Calibration fit: dist = %.3f * rawPred + %.3f\n", myCalibSlope, myCalibOffset);
}

// ─── Adam update helper — plain function (avoids lambda/struct-scope compiler bug) ───
// Called once per parameter group per batch.
void myAdamUpdateGroup(float* w, float* g, float* m, float* v, int n,
                       float lr, float b1, float b2, float eps,
                       float bc1, float bc2, float scale) {
  for (int i = 0; i < n; i++) {
    float gi = g[i] * scale;
    m[i] = b1 * m[i] + (1.0f - b1) * gi;
    v[i] = b2 * v[i] + (1.0f - b2) * gi * gi;
    float mh = m[i] / bc1;
    float vh = v[i] / bc2;
    w[i] -= lr * mh / (sqrtf(vh) + eps);
  }
}

// ─── Struct for training image list ──────────────────────────────
struct MyImgEntry { String path; int label; };

void myFreeImageCache(std::vector<MyImgEntry>& cache) { cache.clear(); cache.shrink_to_fit(); }

// ─── Training action ──────────────────────────────────────────────
void myActionTrain() {
  Serial.println("=== TRAINING START (regression v004) ===");
  u8g2.firstPage(); do { u8g2.drawStr(0,10,"Training..."); } while(u8g2.nextPage());

  // Enumerate images from SD
  std::vector<MyImgEntry> myTrainingData;
  std::vector<MyImgEntry> myValidationData;

  for (int labelIdx = 0; labelIdx < myCfg.numLabels; labelIdx++) {
    String lbl     = myCfg.classLabels[labelIdx];
    String dirPath = "/images/" + lbl;
    if (!mySDavailable || !SD.exists(dirPath.c_str())) continue;
    File root = SD.open(dirPath.c_str());
    if (!root || !root.isDirectory()) continue;
    std::vector<String> files;
    File entry;
    while ((entry = root.openNextFile())) {
      String n = String(entry.name());
      if (n.endsWith(".jpg") || n.endsWith(".JPG")) files.push_back(dirPath + "/" + n);
      entry.close();
    }
    root.close();
    // Split: last validationImages per class → validation, rest → training
    int valN = min((int)files.size(), myCfg.validationImages);
    int trainN = (int)files.size() - valN;
    for (int i = 0; i < trainN; i++) myTrainingData.push_back({files[i], labelIdx});
    for (int i = trainN; i < (int)files.size(); i++) myValidationData.push_back({files[i], labelIdx});
  }

  if (myTrainingData.empty()) {
    Serial.println("ERROR: No training images found on SD!");
    myFreeImageCache(myTrainingData); myFreeImageCache(myValidationData);
    myResetMenuState(); return;
  }
  Serial.printf("Train: %d  Val: %d\n", (int)myTrainingData.size(), (int)myValidationData.size());

  myAllocateMemory(true);
  myAdamStep = 0;

  // Build distance targets from config (evenly spaced normalised values)
  myBuildDistTargets();

  const float myMseWeight       = 1.0f;
  const float myConfLossWeightL = myConfLossWeight;
  const int   myFlat            = CONV2_OUTPUT_SIZE * CONV2_OUTPUT_SIZE * CONV2_FILTERS;
  const float myAdam_lr  = LEARNING_RATE;
  const float myAdam_b1  = 0.9f, myAdam_b2 = 0.999f, myAdam_eps = 1e-8f;

  for (int epoch = 0; epoch < TARGET_EPOCHS; epoch++) {
    // Abort on single-tap (save+exit) or 'l' serial
    if (myPeekTouchAction() == 1) {
      Serial.println("Training: touch-abort → saving weights");
      break;
    }
    if (Serial.available()) {
      char c = Serial.peek();
      if (c == 'l' || c == 'L') { Serial.read(); Serial.println("Training aborted via serial"); break; }
    }

    // Shuffle
    for (int i = (int)myTrainingData.size()-1; i > 0; i--) {
      int j = random(0, i+1);
      MyImgEntry tmp = myTrainingData[i]; myTrainingData[i] = myTrainingData[j]; myTrainingData[j] = tmp;
    }

    float myEpochLoss = 0; int myEpochCount = 0;

    // Mini-batch loop
    for (int batchStart = 0; batchStart < (int)myTrainingData.size(); batchStart += BATCH_SIZE) {
      // Zero gradients
      memset(myConv1_w_grad,  0, CONV1_WEIGHTS*sizeof(float));
      memset(myConv1_b_grad,  0, CONV1_FILTERS*sizeof(float));
      memset(myConv2_w_grad,  0, CONV2_WEIGHTS*sizeof(float));
      memset(myConv2_b_grad,  0, CONV2_FILTERS*sizeof(float));
      memset(myOutput_w_grad, 0, OUTPUT_WEIGHTS*sizeof(float));
      memset(myOutput_b_grad, 0, NUM_CLASSES*sizeof(float));

      int batchActual = 0;

      for (int bi = 0; bi < BATCH_SIZE && (batchStart+bi) < (int)myTrainingData.size(); bi++) {
        auto& img = myTrainingData[batchStart+bi];
        if (!myLoadImageFromFile(img.path.c_str(), myInputBuffer)) continue;
        myForwardPass(myInputBuffer);

        // v003 CHANGE-1: blank class uses distTarget=0.0f (not 0.5f)
        bool myIsBlank = (img.label == 0);
        float myDTarget   = myIsBlank ? 0.0f : myDistTarget[img.label - 1];
        float myConfTarget = myIsBlank ? 0.0f : 1.0f;

        // Losses
        float myDistErr   = myDistPred - myDTarget;
        float myConfErr   = myConfPred - myConfTarget;
        float myLossItem  = myMseWeight * myDistErr * myDistErr
                          + myConfLossWeightL * myConfErr * myConfErr;
        myEpochLoss += myLossItem; myEpochCount++;

        // Output gradients (dist: linear dL/do = 2*(pred-target); conf: after sigmoid)
        float myDistGrad = 2.0f * myMseWeight * myDistErr;          // neuron 0
        float myConfGrad = 2.0f * myConfLossWeightL * myConfErr     // neuron 1
                           * myConfPred * (1.0f - myConfPred);      // sigmoid deriv

        // Accumulate output weight and bias gradients
        for (int i = 0; i < myFlat; i++) {
          float myFlatVal = myConv2_output[i];
          myOutput_w_grad[i * NUM_CLASSES + 0] += myDistGrad * myFlatVal;
          myOutput_w_grad[i * NUM_CLASSES + 1] += myConfGrad * myFlatVal;
          myDense_grad[i] = myDistGrad * myOutput_w[i*NUM_CLASSES+0]
                          + myConfGrad * myOutput_w[i*NUM_CLASSES+1];
        }
        myOutput_b_grad[0] += myDistGrad;
        myOutput_b_grad[1] += myConfGrad;

        // Backprop through Conv2 (ReLU gate)
        for (int i = 0; i < myFlat; i++)
          myConv2_grad[i] = (myConv2_output[i] > 0) ? myDense_grad[i] : 0;

        // Conv2 weight gradients
        for (int f=0; f<CONV2_FILTERS; f++) {
          for (int y=0; y<CONV2_OUTPUT_SIZE; y++) {
            for (int x=0; x<CONV2_OUTPUT_SIZE; x++) {
              float myGrad = myConv2_grad[(y*CONV2_OUTPUT_SIZE+x)*CONV2_FILTERS+f];
              for (int ky=0; ky<3; ky++) {
                for (int kx=0; kx<3; kx++) {
                  int sy=y+ky, sx=x+kx;
                  for (int inf=0; inf<CONV1_FILTERS; inf++) {
                    int wIdx = ((ky*3+kx)*CONV1_FILTERS+inf)*CONV2_FILTERS+f;
                    myConv2_w_grad[wIdx] += myGrad * myPool1_output[(sy*POOL1_OUTPUT_SIZE+sx)*CONV1_FILTERS+inf];
                  }
                }
              }
              myConv2_b_grad[f] += myGrad;
            }
          }
        }

        // Backprop Pool1 (max-pool gradient)
        memset(myPool1_grad, 0, POOL1_OUTPUT_SIZE*POOL1_OUTPUT_SIZE*CONV1_FILTERS*sizeof(float));
        for (int f=0; f<CONV2_FILTERS; f++) {
          for (int y=0; y<CONV2_OUTPUT_SIZE; y++) {
            for (int x=0; x<CONV2_OUTPUT_SIZE; x++) {
              float myConv2G = myConv2_grad[(y*CONV2_OUTPUT_SIZE+x)*CONV2_FILTERS+f];
              // Accumulate into pool1 grad via conv2 weights
              for (int ky=0; ky<3; ky++) {
                for (int kx=0; kx<3; kx++) {
                  int sy=y+ky, sx=x+kx;
                  for (int inf=0; inf<CONV1_FILTERS; inf++) {
                    int wIdx=((ky*3+kx)*CONV1_FILTERS+inf)*CONV2_FILTERS+f;
                    myPool1_grad[(sy*POOL1_OUTPUT_SIZE+sx)*CONV1_FILTERS+inf] += myConv2G * myConv2_w[wIdx];
                  }
                }
              }
            }
          }
        }

        // Max-pool route gradient to winning cell in conv1_output
        memset(myConv1_grad, 0, CONV1_OUTPUT_SIZE*CONV1_OUTPUT_SIZE*CONV1_FILTERS*sizeof(float));
        for (int f=0; f<CONV1_FILTERS; f++) {
          for (int y=0; y<POOL1_OUTPUT_SIZE; y++) {
            for (int x=0; x<POOL1_OUTPUT_SIZE; x++) {
              float poolG = myPool1_grad[(y*POOL1_OUTPUT_SIZE+x)*CONV1_FILTERS+f];
              float mx = -1e9f; int wy=-1, wx=-1;
              for (int py=0; py<2; py++) for (int px=0; px<2; px++) {
                float v = myConv1_output[((y*2+py)*CONV1_OUTPUT_SIZE+(x*2+px))*CONV1_FILTERS+f];
                if (v > mx) { mx=v; wy=y*2+py; wx=x*2+px; }
              }
              if (wy>=0) myConv1_grad[(wy*CONV1_OUTPUT_SIZE+wx)*CONV1_FILTERS+f] += poolG;
            }
          }
        }

        // Conv1 weight gradients (ReLU gated)
        for (int f=0; f<CONV1_FILTERS; f++) {
          for (int y=0; y<CONV1_OUTPUT_SIZE; y++) {
            for (int x=0; x<CONV1_OUTPUT_SIZE; x++) {
              float myGrad = (myConv1_output[(y*CONV1_OUTPUT_SIZE+x)*CONV1_FILTERS+f] > 0)
                           ? myConv1_grad[(y*CONV1_OUTPUT_SIZE+x)*CONV1_FILTERS+f] : 0;
              for (int ky=0; ky<3; ky++) {
                for (int kx=0; kx<3; kx++) {
                  int sy=y+ky, sx=x+kx;
                  for (int c=0; c<INPUT_CHANNELS; c++) {
                    int wIdx = ((ky*3+kx)*INPUT_CHANNELS+c)*CONV1_FILTERS+f;
                    myConv1_w_grad[wIdx] += myGrad * myInputBuffer[(sy*INPUT_SIZE+sx)*INPUT_CHANNELS+c];
                  }
                }
              }
              myConv1_b_grad[f] += myGrad;
            }
          }
        }
        batchActual++;
      } // end mini-batch

      if (batchActual == 0) continue;
      float myScale = 1.0f / batchActual;
      myAdamStep++;

      // Adam update — call plain helper (no lambda; avoids ESP32 gcc struct-scope bug)
      float myBc1 = 1.0f - powf(myAdam_b1, myAdamStep);
      float myBc2 = 1.0f - powf(myAdam_b2, myAdamStep);
      myAdamUpdateGroup(myConv1_w,  myConv1_w_grad,  myConv1_w_m,  myConv1_w_v,  CONV1_WEIGHTS,  myAdam_lr, myAdam_b1, myAdam_b2, myAdam_eps, myBc1, myBc2, myScale);
      myAdamUpdateGroup(myConv1_b,  myConv1_b_grad,  myConv1_b_m,  myConv1_b_v,  CONV1_FILTERS,  myAdam_lr, myAdam_b1, myAdam_b2, myAdam_eps, myBc1, myBc2, myScale);
      myAdamUpdateGroup(myConv2_w,  myConv2_w_grad,  myConv2_w_m,  myConv2_w_v,  CONV2_WEIGHTS,  myAdam_lr, myAdam_b1, myAdam_b2, myAdam_eps, myBc1, myBc2, myScale);
      myAdamUpdateGroup(myConv2_b,  myConv2_b_grad,  myConv2_b_m,  myConv2_b_v,  CONV2_FILTERS,  myAdam_lr, myAdam_b1, myAdam_b2, myAdam_eps, myBc1, myBc2, myScale);
      myAdamUpdateGroup(myOutput_w, myOutput_w_grad, myOutput_w_m, myOutput_w_v, OUTPUT_WEIGHTS, myAdam_lr, myAdam_b1, myAdam_b2, myAdam_eps, myBc1, myBc2, myScale);
      myAdamUpdateGroup(myOutput_b, myOutput_b_grad, myOutput_b_m, myOutput_b_v, NUM_CLASSES,    myAdam_lr, myAdam_b1, myAdam_b2, myAdam_eps, myBc1, myBc2, myScale);
    } // end batches

    float myEpochAvgLoss = myEpochCount > 0 ? myEpochLoss / myEpochCount : 0;
    Serial.printf("TRAIN_EPOCH:%d,%.4f,%.4f\n", epoch, myEpochAvgLoss, 0.0f);
    Serial.printf("Epoch %d/%d loss=%.4f\n", epoch+1, TARGET_EPOCHS, myEpochAvgLoss);

    // OLED update every 5 epochs
    if (epoch % 5 == 0) {
      u8g2.firstPage();
      do {
        u8g2.setFont(u8g2_font_6x10_tf);
        u8g2.drawStr(0,10,"Training");
        u8g2.drawStr(0,22,("E:"+String(epoch+1)+"/"+String(TARGET_EPOCHS)).c_str());
        u8g2.drawStr(0,34,("L:"+String(myEpochAvgLoss,3)).c_str());
      } while(u8g2.nextPage());
    }
  } // end epochs

  // Validation pass + calibration (v003 CHANGE-3c)
  if (!myValidationData.empty()) {
    float myValLoss = 0; int myValCount = 0;
    for (int vi = 0; vi < (int)myValidationData.size(); vi++) {
      const MyImgEntry& vitem = myValidationData[vi];
      if (!myLoadImageFromFile(vitem.path.c_str(), myInputBuffer)) continue;
      myForwardPass(myInputBuffer);
      bool vIsBlank = (vitem.label == 0);
      float vDTarget  = vIsBlank ? 0.0f : myDistTarget[vitem.label - 1];
      float vDE = myDistPred - vDTarget;
      float vCE = myConfPred - (vIsBlank ? 0.0f : 1.0f);
      myValLoss += vDE*vDE + vCE*vCE;
      myValCount++;
    }
    Serial.printf("Validation MAE=%.4f  n=%d\n",
      myValCount>0 ? myValLoss/myValCount : 0.0f, myValCount);
  }

  // Calibration pass (v003 CHANGE-3c) — run forward on all training images per dist class
  {
    float myCalibSumPred[8] = {};
    int   myCalibCount[8]   = {};
    for (int ci2 = 0; ci2 < (int)myTrainingData.size(); ci2++) {
      const MyImgEntry& citem = myTrainingData[ci2];
      if (citem.label == 0) continue;  // skip blank
      if (!myLoadImageFromFile(citem.path.c_str(), myInputBuffer)) continue;
      myForwardPass(myInputBuffer);
      int ci = citem.label - 1;
      myCalibSumPred[ci] += myDistPred;
      myCalibCount[ci]++;
    }
    float myAnchorRaw[8]; float myAnchorDist[8]; int myAnchorN = 0;
    for (int ci = 0; ci < NUM_DIST_CLASSES; ci++) {
      if (myCalibCount[ci] > 0) {
        myAnchorRaw[myAnchorN]  = myCalibSumPred[ci] / myCalibCount[ci];
        myAnchorDist[myAnchorN] = myDistTarget[ci] * myDistMax;
        Serial.printf("  Calib anchor: %s meanRaw=%.4f real=%.1fcm\n",
          myDistLabel[ci].c_str(), myAnchorRaw[myAnchorN], myAnchorDist[myAnchorN]);
        myAnchorN++;
      }
    }
    myFitCalibration(myAnchorRaw, myAnchorDist, myAnchorN);
  }

  myFreeImageCache(myTrainingData);
  myFreeImageCache(myValidationData);
  mySaveWeights();
  myWeightsTrained = true;
  Serial.println("=== TRAINING COMPLETE ===");
  Serial.printf("CALIB_STATUS:slope=%.4f,offset=%.4f,ready=%d\n",
    myCalibSlope, myCalibOffset, (int)myCalibReady);
  u8g2.firstPage(); do { u8g2.drawStr(0,10,"Train done!"); u8g2.drawStr(0,22,"Weights saved"); } while(u8g2.nextPage());
  delay(2000);
}

// ─── Inference action ─────────────────────────────────────────────
void myActionInfer() {
  if (!myWeightsTrained) {
    Serial.println("ERR: No trained weights — train or load first");
    u8g2.firstPage(); do { u8g2.drawStr(0,10,"No weights!"); u8g2.drawStr(0,22,"Train first"); } while(u8g2.nextPage());
    delay(2000); return;
  }
  Serial.println("=== INFERENCE MODE (3-tap to exit) ===");
  myResetTouchState();

  while (true) {
    if (Serial.available()) {
      char c = Serial.read();
      mySerialLineBuf += c;
      if (c == '\n' || c == '\r') {
        String cmd = mySerialLineBuf; cmd.trim(); mySerialLineBuf = "";
        if (cmd == "l" || cmd == "L") { myResetMenuState(); return; }
        if (cmd.length() > 0) myHandleStringCommand(cmd);
      }
    }
    if (myCheckTouchInput() == 2) { myResetMenuState(); return; }

    camera_fb_t* fb = esp_camera_fb_get();
    if (!fb) { delay(10); continue; }

    // Decode JPEG → float buffer
    if (myRgbBuffer && fmt2rgb888(fb->buf, fb->len, PIXFORMAT_JPEG, myRgbBuffer)) {
      for (int y=0; y<INPUT_SIZE; y++) {
        for (int x=0; x<INPUT_SIZE; x++) {
          int sy=(int)((y+0.5)*240.0/INPUT_SIZE); if(sy>239)sy=239;
          int sx=(int)((x+0.5)*240.0/INPUT_SIZE); if(sx>239)sx=239;
          int srcIdx=(sy*240+sx)*3, dstIdx=(y*INPUT_SIZE+x)*3;
          myInputBuffer[dstIdx]   = myRgbBuffer[srcIdx]   / 255.0f;
          myInputBuffer[dstIdx+1] = myRgbBuffer[srcIdx+1] / 255.0f;
          myInputBuffer[dstIdx+2] = myRgbBuffer[srcIdx+2] / 255.0f;
        }
      }
      myForwardPass(myInputBuffer);

      // v003 CHANGE-3d: calibrated distance
      float myRealDist = myCalibReady
        ? (myCalibSlope * myDistPred + myCalibOffset)
        : (myDistPred * myDistMax);
      bool myObjectDetected = (myConfPred >= myConfThreshold);

      // WebSerial INFER output: "INFER:<dist_cm>,<conf>"
      Serial.printf("INFER:%.3f,%.4f\n", myRealDist, myConfPred);

      // OLED display
      if (myObjectDetected) {
        u8g2.firstPage();
        do {
          u8g2.setFont(u8g2_font_6x10_tf);
          u8g2.drawStr(0,10,("D:" + String(myRealDist,1) + "cm").c_str());
          u8g2.drawStr(0,22,("C:" + String(myConfPred,2)).c_str());
          u8g2.drawStr(0,34,"3tap=exit");
        } while(u8g2.nextPage());
      } else {
        u8g2.firstPage();
        do {
          u8g2.setFont(u8g2_font_6x10_tf);
          u8g2.drawStr(0,10,"No object");
          u8g2.drawStr(0,22,("C:" + String(myConfPred,2)).c_str());
          u8g2.drawStr(0,34,"3tap=exit");
        } while(u8g2.nextPage());
      }

      if (myHeatmapEnabled) mySendHeatmap();

      if (myCamStreaming) myCamCaptureSend();
    }
    esp_camera_fb_return(fb);
    delay(100);
  }
}

void myBase64SendFrame(camera_fb_t* fb) {
  size_t b64Len = 0;
  size_t maxOut = ((fb->len + 2) / 3) * 4 + 1;
  uint8_t* b64  = (uint8_t*)malloc(maxOut);
  if (!b64) return;
  mbedtls_base64_encode(b64, maxOut, &b64Len, fb->buf, fb->len);
  Serial.print("FRAME_B64:");
  Serial.write(b64, b64Len);
  Serial.println();
  free(b64);
}

void myCamCaptureSend() {
  camera_fb_t* fb = esp_camera_fb_get();
  if (!fb) return;
  Serial.println("CAM_JPEG_START");
  size_t maxOut = ((fb->len + 2) / 3) * 4 + 1;
  uint8_t* b64 = (uint8_t*)malloc(maxOut);
  if (b64) {
    size_t b64Len = 0;
    mbedtls_base64_encode(b64, maxOut, &b64Len, fb->buf, fb->len);
    const int chunkSz = 512;
    for (size_t i=0; i<b64Len; i+=chunkSz) {
      size_t end = min(i+(size_t)chunkSz, b64Len);
      Serial.print("CAM_JPEG:"); Serial.write(b64+i, end-i); Serial.println();
    }
    free(b64);
  }
  Serial.println("CAM_JPEG_END");
  esp_camera_fb_return(fb);
}

void myActionWebStream() {
  Serial.println("WebStream: 'l'=exit");
  myResetTouchState();
  while (true) {
    if (Serial.available()) { char c=Serial.read(); if(c=='l'||c=='L'){myResetMenuState();return;} }
    if (myCheckTouchInput() == 2) { myResetMenuState(); return; }
    camera_fb_t* fb = esp_camera_fb_get();
    if (fb) { myBase64SendFrame(fb); esp_camera_fb_return(fb); }
    delay(1000);
  }
}

void mySendHeatmap() {
  if (!myConv2_output) return;
  int rows=CONV2_OUTPUT_SIZE, cols=CONV2_OUTPUT_SIZE, filters=CONV2_FILTERS;
  int nPixels=rows*cols;
  uint8_t* pix=(uint8_t*)malloc(nPixels); if(!pix) return;
  float minV=1e9, maxV=-1e9;
  for(int i=0;i<nPixels;i++){
    float mx=myConv2_output[i];
    for(int f=1;f<filters;f++) mx=max(mx,myConv2_output[f*nPixels+i]);
    minV=min(minV,mx); maxV=max(maxV,mx);
  }
  for(int i=0;i<nPixels;i++){
    float mx=myConv2_output[i];
    for(int f=1;f<filters;f++) mx=max(mx,myConv2_output[f*nPixels+i]);
    pix[i]=(maxV>minV)?(uint8_t)(255*(mx-minV)/(maxV-minV)):0;
  }
  size_t maxOut=((nPixels+2)/3)*4+1;
  uint8_t* b64=(uint8_t*)malloc(maxOut);
  if(b64){size_t b64Len=0; mbedtls_base64_encode(b64,maxOut,&b64Len,pix,nPixels); Serial.printf("HEATMAP:%dx%d:",rows,cols); Serial.write(b64,b64Len); Serial.println(); free(b64);}
  free(pix);
}

void myResetMenuState() {
  myMenuIndex  = 1;
  myIsSelected = false;
  myResetTouchState();
  myDrawMenu();
}

void myDrawMenu() {
  u8g2.firstPage();
  do {
    u8g2.setFont(u8g2_font_6x10_tf);
    if (!myIsSelected) {
      String label;
      if (myMenuIndex <= myCfg.numLabels)
        label = "Col " + myCfg.classLabels[myMenuIndex-1].substring(0,6);
      else if (myMenuIndex == myCfg.numLabels+1) label = "Train";
      else label = "Infer";
      u8g2.drawStr(0, 10, (">" + label).c_str());
      u8g2.drawStr(0, 22, "tap=next");
      u8g2.drawStr(0, 34, "3tap=select");
    } else {
      u8g2.drawStr(0, 10, "Running...");
      u8g2.drawStr(0, 22, "3tap=exit");
    }
  } while (u8g2.nextPage());
}

void myDispatchByIndex(int idx) {
  if (idx <= myCfg.numLabels) {
    myIsSelected=true; myDrawMenu(); myActionCollect(idx-1); myResetMenuState();
  } else if (idx == myCfg.numLabels+1) {
    myIsSelected=true; myDrawMenu(); myActionTrain();         myResetMenuState();
  } else {
    myIsSelected=true; myDrawMenu(); myActionInfer();         myResetMenuState();
  }
}

void myDispatchMenuCmd(const String& cmd) {
  if (cmd=="TRAIN")  { myIsSelected=true; myDrawMenu(); myActionTrain();     myResetMenuState(); return; }
  if (cmd=="INFER")  { myIsSelected=true; myDrawMenu(); myActionInfer();     myResetMenuState(); return; }
  if (cmd=="STREAM") { myIsSelected=true; myDrawMenu(); myActionWebStream(); myResetMenuState(); return; }
  for (int i=0; i<myCfg.numLabels; i++) {
    if (cmd==String("COLLECT")+i) {
      myIsSelected=true; myDrawMenu(); myActionCollect(i); myResetMenuState(); return;
    }
  }
}

void myHandleMenuNavigation() {
  while (Serial.available()) {
    char c = Serial.read();
    if (c=='\n'||c=='\r') {
      if (mySerialLineBuf.length() > 0) {
        myHandleStringCommand(mySerialLineBuf);
        mySerialLineBuf = "";
      }
    } else {
      if (mySerialLineBuf.length() < 4096) mySerialLineBuf += c;
      else { Serial.println("ERR:line too long"); mySerialLineBuf = ""; }
    }
  }
  int touchAction = myCheckTouchInput();
  if (myIsSelected) return;
  if (touchAction == 1) {
    myMenuIndex++;
    if (myMenuIndex > myTotalItems) myMenuIndex = 1;
    myDrawMenu();
  } else if (touchAction == 2) {
    myDispatchByIndex(myMenuIndex);
  }
}

void mySdReadJpeg(const String& path) {
  if (!mySDavailable) { Serial.println("ERR:SD not available"); return; }
  File f = SD.open(myNormPath(path).c_str(), FILE_READ);
  if (!f) { Serial.println("ERR:Cannot open file"); return; }
  size_t sz=f.size();
  uint8_t* buf=(uint8_t*)ps_malloc(sz); if(!buf){f.close();Serial.println("ERR:PSRAM alloc failed");return;}
  f.read(buf,sz); f.close();
  size_t maxOut=((sz+2)/3)*4+1;
  uint8_t* b64=(uint8_t*)malloc(maxOut); if(!b64){free(buf);Serial.println("ERR:b64 alloc");return;}
  size_t b64Len=0; mbedtls_base64_encode(b64,maxOut,&b64Len,buf,sz); free(buf);
  Serial.println("SD_JPEG_START");
  const int chunkSz=512;
  for(size_t i=0;i<b64Len;i+=chunkSz){size_t end=min(i+(size_t)chunkSz,b64Len); Serial.print("SD_JPEG:"); Serial.write(b64+i,end-i); Serial.println();}
  free(b64);
  Serial.println("SD_JPEG_END");
}

void mySdReadBinaryHead(const String& path, uint16_t numBytes) {
  if (!mySDavailable) { Serial.println("ERR:SD not available"); return; }
  File f = SD.open(myNormPath(path).c_str(), FILE_READ);
  if (!f) { Serial.println("ERR:Cannot open file"); return; }
  uint8_t buf[256]; int n=0;
  while(f.available()&&n<numBytes&&n<256) buf[n++]=f.read();
  f.close();
  Serial.printf("SD_HEAD_START:%d\n",n);
  for(int i=0;i<n;i++){if(buf[i]>=32&&buf[i]<127)Serial.write(buf[i]);else Serial.print('.');}
  Serial.println(); Serial.println("SD_HEAD_END");
}

bool mySdRemoveDirRecursive(const String& path, int& deleted) {
  File root=SD.open(path.c_str());
  if(!root||!root.isDirectory()) return false;
  File entry;
  while((entry=root.openNextFile())){
    String childPath=path+"/"+entry.name();
    if(entry.isDirectory()){entry.close();mySdRemoveDirRecursive(childPath,deleted);SD.rmdir(childPath.c_str());}
    else{entry.close();if(SD.remove(childPath.c_str()))deleted++;}
  }
  root.close();
  return SD.rmdir(path.c_str());
}

void myHandleStringCommand(const String& rawCmd) {
  String cmd = rawCmd; cmd.trim();
  if (cmd.length() == 0) return;

  // Single-char menu shortcuts
  if (cmd.length()==1) {
    char c = cmd[0];
    if (c>='1' && c<='9') { int idx=c-'0'; if(idx<=myTotalItems) myDispatchByIndex(idx); return; }
    return;
  }

  // Status
  if (cmd=="STATUS") {
    Serial.println("STATUS:OK");
    Serial.printf("STATUS:version=webmcu-regression-%d\n", CURRENT_VERSION);
    Serial.printf("STATUS:numLabels=%d inputSize=%d numDistClasses=%d distMax=%.1f\n",
      myCfg.numLabels, INPUT_SIZE, NUM_DIST_CLASSES, myDistMax);
    Serial.printf("STATUS:SD=%s weightsTrained=%s calibReady=%s\n",
      mySDavailable?"yes":"no", myWeightsTrained?"yes":"no", myCalibReady?"yes":"no");
    if (myCalibReady) Serial.printf("STATUS:calibSlope=%.4f calibOffset=%.4f\n", myCalibSlope, myCalibOffset);
    Serial.println("STATUS:commands=SD_LIST,SD_READ,SD_WRITE,SD_TEXT_WRITE_START/CHUNK/END,SD_JPEG_WRITE_START/CHUNK/END,SD_DELETE,SD_MKDIR,SD_RMDIR,SD_HEAD,SD_JPEG,CAM_CAPTURE,CAM_STREAM,CAM_STREAM_STOP,HEATMAP_ON,HEATMAP_OFF,HEATMAP_STATUS,FILE_SEND_START,FILE_CHUNK,FILE_SEND_END,CALIB_STATUS,DIST_MAX_SET,CONF_THRESHOLD_SET");
    return;
  }

  // Heatmap
  if (cmd=="HEATMAP_ON")     { myHeatmapEnabled=true;  Serial.println("HEATMAP_STATUS:on");  return; }
  if (cmd=="HEATMAP_OFF")    { myHeatmapEnabled=false; Serial.println("HEATMAP_STATUS:off"); return; }
  if (cmd=="HEATMAP_STATUS") { Serial.printf("HEATMAP_STATUS:%s %dx%d\n",myHeatmapEnabled?"on":"off",CONV2_OUTPUT_SIZE,CONV2_OUTPUT_SIZE); return; }

  // Camera
  if (cmd=="CAM_CAPTURE")    { myCamCaptureSend(); return; }
  if (cmd=="CAM_STREAM")     { myCamStreaming=true;  Serial.println("CAM_STREAM:on");  return; }
  if (cmd=="CAM_STREAM_STOP"){ myCamStreaming=false; Serial.println("CAM_STREAM:off"); return; }

  // SD browser
  if (cmd.startsWith("SD_LIST:"))  { mySdListDir(cmd.substring(8)); return; }
  if (cmd.startsWith("SD_READ:"))  { mySdReadText(cmd.substring(8)); return; }
  if (cmd.startsWith("SD_WRITE:")) {
    int c2=cmd.indexOf(':',9); if(c2<0){Serial.println("ERR:SD_WRITE format");return;}
    mySdWriteText(cmd.substring(9,c2),cmd.substring(c2+1)); return;
  }
  if (cmd.startsWith("SD_TEXT_WRITE_START:")) { myTextWritePath=myNormPath(cmd.substring(20)); myTextWriteContent=""; myTextWriteActive=true; Serial.println("OK:SD_TEXT_WRITE_START"); return; }
  if (cmd.startsWith("SD_TEXT_CHUNK:") && myTextWriteActive) { String chunk=cmd.substring(14); chunk.replace("\\n","\n"); myTextWriteContent+=chunk; return; }
  if (cmd=="SD_TEXT_WRITE_END" && myTextWriteActive) { myTextWriteActive=false; mySdWriteText(myTextWritePath,myTextWriteContent); myTextWritePath=""; myTextWriteContent=""; return; }
  if (cmd.startsWith("SD_JPEG:"))  { mySdReadJpeg(cmd.substring(8)); return; }
  if (cmd.startsWith("SD_HEAD:"))  { mySdReadBinaryHead(cmd.substring(8),256); return; }
  if (cmd.startsWith("SD_JPEG_WRITE_START:")) { myJpegWritePath=myNormPath(cmd.substring(20)); myJpegWriteB64=""; myJpegWriteActive=true; Serial.println("OK:SD_JPEG_WRITE_START"); return; }
  if (cmd.startsWith("SD_JPEG_CHUNK:") && myJpegWriteActive) { myJpegWriteB64+=cmd.substring(14); return; }
  if (cmd=="SD_JPEG_WRITE_END" && myJpegWriteActive) {
    myJpegWriteActive=false;
    if(mySDavailable){
      size_t b64Len=myJpegWriteB64.length(), maxBin=(b64Len*3)/4+4; size_t binLen=0;
      uint8_t* bin=(uint8_t*)ps_malloc(maxBin);
      if(bin&&mbedtls_base64_decode(bin,maxBin,&binLen,(const uint8_t*)myJpegWriteB64.c_str(),b64Len)==0){
        myEnsureParentDir(myJpegWritePath);
        File f=SD.open(myJpegWritePath.c_str(),FILE_WRITE);
        if(f){f.write(bin,binLen);f.close();Serial.println("OK:SD_JPEG_WRITE_END");}
        else  Serial.println("ERR:Cannot create JPEG file");
      } else Serial.println("ERR:base64 decode failed");
      if(bin) free(bin);
    } else Serial.println("ERR:SD not available");
    myJpegWritePath=""; myJpegWriteB64=""; return;
  }
  if (cmd.startsWith("SD_DELETE:")) { if(!mySDavailable){Serial.println("ERR:SD not available");return;} String p=myNormPath(cmd.substring(10)); Serial.println(SD.remove(p.c_str())?"OK:deleted":"ERR:delete failed"); return; }
  if (cmd.startsWith("SD_MKDIR:"))  { if(!mySDavailable){Serial.println("ERR:SD not available");return;} SD.mkdir(myNormPath(cmd.substring(9)).c_str()); Serial.println("OK:mkdir"); return; }
  if (cmd.startsWith("SD_RMDIR:"))  { if(!mySDavailable){Serial.println("ERR:SD not available");return;} int deleted=0; mySdRemoveDirRecursive(myNormPath(cmd.substring(9)),deleted); Serial.printf("OK:rmdir deleted=%d\n",deleted); return; }

  // v78: Binary file transfer
  if (cmd.startsWith("FILE_SEND_START:")) {
    if(!mySDavailable){Serial.println("ERR:SD not available");return;}
    int colon=cmd.indexOf(':',16); if(colon<0){Serial.println("ERR:FILE_SEND_START format");return;}
    myFileRecvPath=myNormPath(cmd.substring(16,colon));
    myFileRecvBytes=0;
    if(myFileRecvActive && myFileRecvHandle) myFileRecvHandle.close();
    myEnsureParentDir(myFileRecvPath);
    myFileRecvHandle=SD.open(myFileRecvPath.c_str(),FILE_WRITE);
    if(!myFileRecvHandle){Serial.println("ERR:Cannot open file for write");return;}
    myFileRecvActive=true;
    Serial.println("OK:FILE_READY");
    return;
  }
  if (cmd.startsWith("FILE_CHUNK:") && myFileRecvActive) {
    int c1=cmd.indexOf(':',11); int c2=cmd.indexOf(':',c1+1); int c3=cmd.indexOf(':',c2+1);
    if(c1<0||c2<0||c3<0){Serial.println("ERR:FILE_CHUNK format");return;}
    int idx=cmd.substring(11,c1).toInt();
    uint8_t expectedXor=(uint8_t)strtoul(cmd.substring(c1+1,c2).c_str(),nullptr,16);
    String hex=cmd.substring(c3+1);
    int hexLen=hex.length(); if(hexLen%2!=0){Serial.printf("ERR:CHUNK_RETRY:%d:odd hex\n",idx);return;}
    int byteLen=hexLen/2;
    uint8_t* buf=(uint8_t*)malloc(byteLen); if(!buf){Serial.printf("ERR:CHUNK_RETRY:%d:malloc\n",idx);return;}
    uint8_t xorCalc=0;
    for(int i=0;i<byteLen;i++){buf[i]=(uint8_t)strtoul(hex.substring(i*2,i*2+2).c_str(),nullptr,16);xorCalc^=buf[i];}
    if(xorCalc!=expectedXor){free(buf);Serial.printf("ERR:CHUNK_RETRY:%d:checksum\n",idx);return;}
    myFileRecvHandle.write(buf,byteLen); free(buf);
    myFileRecvBytes+=byteLen;
    Serial.printf("OK:CHUNK_ACK:%d\n",idx);
    return;
  }
  if (cmd.startsWith("FILE_SEND_END:") && myFileRecvActive) {
    long expected=cmd.substring(14).toInt();
    myFileRecvHandle.close(); myFileRecvActive=false;
    if(myFileRecvBytes==expected){
      Serial.printf("OK:FILE_DONE:%s:%ld\n",myFileRecvPath.c_str(),myFileRecvBytes);
    } else {
      Serial.printf("ERR:FILE_SIZE_MISMATCH:got=%ld expected=%ld\n",myFileRecvBytes,expected);
    }
    myFileRecvPath=""; myFileRecvBytes=0;
    return;
  }

  // [INJECT-7] Regression-specific commands
  // CALIB_STATUS — report current calibration parameters
  if (cmd=="CALIB_STATUS") {
    Serial.printf("CALIB_STATUS:ready=%d slope=%.6f offset=%.6f distMax=%.2f\n",
      (int)myCalibReady, myCalibSlope, myCalibOffset, myDistMax);
    return;
  }
  // DIST_MAX_SET:<value> — update distMax at runtime
  if (cmd.startsWith("DIST_MAX_SET:")) {
    float newMax = cmd.substring(13).toFloat();
    if (newMax > 0) { myCfg.distMax = newMax; Serial.printf("OK:distMax=%.2f\n", newMax); }
    else Serial.println("ERR:invalid distMax value");
    return;
  }
  // CONF_THRESHOLD_SET:<value> — update confidence threshold at runtime
  if (cmd.startsWith("CONF_THRESHOLD_SET:")) {
    float newThr = cmd.substring(19).toFloat();
    if (newThr >= 0 && newThr <= 1.0f) { myCfg.confThreshold = newThr; Serial.printf("OK:confThreshold=%.3f\n", newThr); }
    else Serial.println("ERR:confThreshold must be 0..1");
    return;
  }

  Serial.printf("ERR:unknown command: %s\n", cmd.c_str());
}
